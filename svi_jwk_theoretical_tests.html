1. Overview of the two theoretical tests

TestQuestion answeredWhy it is conceptual soundness
Test 1 — JWK-to-raw-SVI parameterisation integrity Does the JWK parameter set map into a valid raw-SVI slice, and is the conversion stable except where the theory itself becomes degenerate? This tests the mathematical design claim that JWK is a safe reparameterisation of raw SVI.
Test 2 — Surface-level calendar construction If two SVI-JWK tenor slices are individually valid, does the chosen interpolation rule preserve total-variance monotonicity between tenors? This tests whether the model is sound as a full volatility surface, not merely as independent slices.

8. Embedded executable Python code

File: svi_jwk_theoretical_conceptual_tests.py
#!/usr/bin/env python3
"""
SVI-JWK theoretical conceptual soundness tests.

This script implements two model-design tests for an SVI-JWK volatility
surface model:

Test 1
------
JWK-to-raw-SVI parameterisation integrity and admissible-domain equivalence.

Test 2
------
Surface-level calendar construction and interpolation preservation test.

The implementation is intentionally transparent and dependency-light:
Python standard library + NumPy only.

Important convention note
-------------------------
Different institutions sometimes store the JWK "kappa" parameter using a
rescaled convention. This script supports two input conventions:

1. raw
   The input kappa is used directly in the JWK -> raw inverse formula.

2. stored_div_sqrtT
   The input kappa is assumed to have been stored as kappa_raw / sqrt(T);
   the script multiplies by sqrt(T) before applying the inverse formula.

The JWK -> raw inverse used here is:

    w_t   = nu_t * T
    b     = sqrt(w_t) * (c_t + p_t) / 2
    rho   = (c_t - p_t) / (c_t + p_t)
    beta  = (c_t - p_t - 4 psi_t) / (c_t + p_t)
    gamma = sqrt(w_t) * (c_t + p_t)
            / (4 * (2 sqrt(w_t) kappa_raw + psi_t^2))
    m     = beta * (1 - beta^2) * gamma
    sigma = (1 - beta^2)^(3/2) * gamma
    a     = w_t - b * {-rho*m + sqrt(m^2 + sigma^2)}

The beta=0 and |beta|=1 limits are handled explicitly. At |beta|=1 the
mapping is degenerate: m=0 and sigma=0, and kappa is not identifiable from
the resulting raw-SVI slice. This is reported as a boundary limitation rather
than hidden as a numerical error.

The raw-SVI admissibility checks follow the standard raw-SVI minimum
requirements:

    b >= 0
    |rho| <= 1
    sigma >= 0
    a + b*sigma*sqrt(1-rho^2) >= 0

References
----------
- Gatheral, J. (2006). The Volatility Surface. Wiley.
- Gatheral, J. and Jacquier, A. (2014). Arbitrage-free SVI volatility surfaces.
  Quantitative Finance 14(1), 59-71.
- Lee, R. W. (2004). The moment formula for implied volatility at extreme
  strikes. Mathematical Finance 14(3), 469-480.
- Roper, M. (2010). Arbitrage Free Implied Volatility Surfaces.
- QuantLib SviInterpolatedSmileSection documentation, used only as a
  standard industry codebase reference for SVI-style smile construction.
"""

from __future__ import annotations

import argparse
import csv
import json
import math
import os
from dataclasses import dataclass
from typing import Dict, Iterable, List, Optional, Tuple

import numpy as np


DOMAIN_REGULAR = "regular"
DOMAIN_AT_BOUNDARY = "at_boundary"
DOMAIN_OUTSIDE = "outside_admissible_region"


@dataclass(frozen=True)
class RawSVI:
    a: float
    b: float
    rho: float
    m: float
    sigma: float


@dataclass(frozen=True)
class JWK:
    case_id: str
    slice_id: str
    tenor: float
    nu: float
    psi: float
    kappa: float
    p: float
    c: float


@dataclass(frozen=True)
class Config:
    kappa_convention: str = "raw"
    boundary_tol: float = 1e-10
    roundtrip_tol: float = 1e-8
    k_min: float = -3.0
    k_max: float = 3.0
    k_points: int = 301
    t_interpolation_points: int = 25
    calendar_tol: float = 1e-10
    interpolation_rule: str = "linear_jwk"


def raw_svi_total_variance(k: np.ndarray, raw: RawSVI) -> np.ndarray:
    u = np.asarray(k, dtype=float) - raw.m
    return raw.a + raw.b * (raw.rho * u + np.sqrt(u * u + raw.sigma * raw.sigma))


def classify_raw_svi(raw: RawSVI, tol: float = 1e-10) -> Dict[str, object]:
    details: List[str] = []
    outside = False
    boundary = False

    if raw.b < -tol:
        outside = True
        details.append("b < 0")
    elif raw.b <= tol:
        boundary = True
        details.append("b at zero boundary")

    if abs(raw.rho) > 1.0 + tol:
        outside = True
        details.append("|rho| > 1")
    elif abs(raw.rho) >= 1.0 - tol:
        boundary = True
        details.append("|rho| at unit boundary")

    if raw.sigma < -tol:
        outside = True
        details.append("sigma < 0")
    elif raw.sigma <= tol:
        boundary = True
        details.append("sigma at zero boundary")

    one_minus = max(0.0, 1.0 - raw.rho * raw.rho)
    min_tv_formula = raw.a + raw.b * raw.sigma * math.sqrt(one_minus)
    if min_tv_formula < -tol:
        outside = True
        details.append("minimum total variance < 0")
    elif min_tv_formula <= tol:
        boundary = True
        details.append("minimum total variance at zero boundary")

    if outside:
        status = DOMAIN_OUTSIDE
    elif boundary:
        status = DOMAIN_AT_BOUNDARY
    else:
        status = DOMAIN_REGULAR

    return {
        "domain_status": status,
        "details": "; ".join(details) if details else "OK",
        "min_total_variance_formula": min_tv_formula,
    }


def _kappa_to_raw(kappa: float, tenor: float, convention: str) -> float:
    if convention == "raw":
        return kappa
    if convention == "stored_div_sqrtT":
        return kappa * math.sqrt(tenor)
    raise ValueError("kappa_convention must be 'raw' or 'stored_div_sqrtT'")


def _kappa_from_raw(kappa_raw: float, tenor: float, convention: str) -> float:
    if convention == "raw":
        return kappa_raw
    if convention == "stored_div_sqrtT":
        return kappa_raw / math.sqrt(tenor)
    raise ValueError("kappa_convention must be 'raw' or 'stored_div_sqrtT'")


def jwk_to_raw_svi(jwk: JWK, config: Config) -> Tuple[Optional[RawSVI], Dict[str, object]]:
    """
    Convert one SVI-JWK slice to raw SVI using the stated converter convention.

    Returns (raw, diagnostic). raw is None if the converter is outside its
    mathematical domain.
    """
    T = jwk.tenor
    if T <= 0.0:
        return None, {"conversion_status": "FAIL", "reason": "tenor <= 0"}

    w_t = jwk.nu * T
    if w_t < 0.0:
        return None, {"conversion_status": "FAIL", "reason": "nu*T < 0"}

    sum_slopes = jwk.c + jwk.p
    if sum_slopes <= 0.0:
        return None, {"conversion_status": "FAIL", "reason": "c+p <= 0"}

    sqrt_w = math.sqrt(max(w_t, 0.0))
    b = sqrt_w * sum_slopes / 2.0
    rho = (jwk.c - jwk.p) / sum_slopes
    beta = (jwk.c - jwk.p - 4.0 * jwk.psi) / sum_slopes

    if abs(rho) > 1.0 + config.boundary_tol:
        return None, {
            "conversion_status": "FAIL",
            "reason": "|rho| > 1 implied by p,c",
            "beta": beta,
            "rho": rho,
        }

    if abs(beta) > 1.0 + config.boundary_tol:
        return None, {
            "conversion_status": "FAIL",
            "reason": "|beta| > 1; inverse gives imaginary sigma",
            "beta": beta,
            "rho": rho,
        }

    # Boundary collapse at |beta| = 1. gamma is not identifiable from raw output.
    if abs(abs(beta) - 1.0) <= config.boundary_tol:
        raw = RawSVI(a=w_t, b=b, rho=rho, m=0.0, sigma=0.0)
        return raw, {
            "conversion_status": "BOUNDARY_DEGENERATE",
            "reason": "|beta|=1; m=sigma=0 and kappa is not identifiable from raw slice",
            "beta": beta,
            "rho": rho,
            "gamma": "",
        }

    kappa_raw = _kappa_to_raw(jwk.kappa, T, config.kappa_convention)
    denom = 2.0 * sqrt_w * kappa_raw + jwk.psi * jwk.psi
    if denom <= config.boundary_tol:
        return None, {
            "conversion_status": "FAIL",
            "reason": "gamma denominator <= tolerance",
            "beta": beta,
            "rho": rho,
            "denominator": denom,
        }

    gamma = sqrt_w * sum_slopes / (4.0 * denom)

    if abs(beta) <= config.boundary_tol:
        m = 0.0
        sigma = gamma
    else:
        m = beta * (1.0 - beta * beta) * gamma
        sigma = (1.0 - beta * beta) ** 1.5 * gamma

    a = w_t - b * (-rho * m + math.sqrt(m * m + sigma * sigma))
    raw = RawSVI(a=a, b=b, rho=rho, m=m, sigma=sigma)

    return raw, {
        "conversion_status": "OK",
        "reason": "regular conversion",
        "beta": beta,
        "rho": rho,
        "gamma": gamma,
    }


def raw_to_jwk_consistent(
    raw: RawSVI,
    case_id: str,
    slice_id: str,
    tenor: float,
    config: Config,
) -> Tuple[Optional[JWK], Dict[str, object]]:
    """
    Convert raw SVI back to JWK under the same converter algebra.

    At sigma=0 or |beta|=1, kappa is generally not identifiable from the raw
    slice. The function reports this explicitly.
    """
    if tenor <= 0.0:
        return None, {"reverse_status": "FAIL", "reason": "tenor <= 0"}

    w0 = float(raw_svi_total_variance(np.array([0.0]), raw)[0])
    if w0 <= config.boundary_tol:
        return None, {"reverse_status": "FAIL", "reason": "ATM total variance <= tolerance"}

    sqrt_w = math.sqrt(w0)
    root0 = math.sqrt(raw.m * raw.m + raw.sigma * raw.sigma)
    if root0 <= config.boundary_tol:
        return None, {
            "reverse_status": "BOUNDARY_DEGENERATE",
            "reason": "m=sigma=0; ATM derivative and kappa are not uniquely recoverable",
        }

    nu = w0 / tenor
    psi = raw.b / (2.0 * sqrt_w) * (raw.rho - raw.m / root0)
    p = raw.b / sqrt_w * (1.0 - raw.rho)
    c = raw.b / sqrt_w * (1.0 + raw.rho)

    sum_slopes = p + c
    if sum_slopes <= config.boundary_tol:
        return None, {"reverse_status": "FAIL", "reason": "p+c <= tolerance"}

    beta = (c - p - 4.0 * psi) / sum_slopes
    if abs(abs(beta) - 1.0) <= config.boundary_tol:
        return None, {
            "reverse_status": "BOUNDARY_DEGENERATE",
            "reason": "|beta|=1; kappa is not identifiable",
        }

    if abs(beta) <= config.boundary_tol:
        gamma = raw.sigma
    else:
        gamma = raw.m / (beta * (1.0 - beta * beta))

    if gamma <= config.boundary_tol:
        return None, {"reverse_status": "FAIL", "reason": "gamma <= tolerance"}

    kappa_raw = (sqrt_w * sum_slopes / (4.0 * gamma) - psi * psi) / (2.0 * sqrt_w)
    kappa = _kappa_from_raw(kappa_raw, tenor, config.kappa_convention)

    return JWK(case_id, slice_id, tenor, nu, psi, kappa, p, c), {
        "reverse_status": "OK",
        "reason": "regular reverse conversion",
        "beta": beta,
        "gamma": gamma,
    }


def load_jwk_csv(path: str) -> List[JWK]:
    required = {"case_id", "slice_id", "tenor", "nu", "psi", "kappa", "p", "c"}
    out: List[JWK] = []
    with open(path, "r", newline="", encoding="utf-8") as f:
        reader = csv.DictReader(f)
        missing = required.difference(reader.fieldnames or [])
        if missing:
            raise ValueError(f"Missing required CSV columns: {sorted(missing)}")
        for line_no, row in enumerate(reader, start=2):
            out.append(JWK(
                case_id=row["case_id"].strip(),
                slice_id=row["slice_id"].strip(),
                tenor=float(row["tenor"]),
                nu=float(row["nu"]),
                psi=float(row["psi"]),
                kappa=float(row["kappa"]),
                p=float(row["p"]),
                c=float(row["c"]),
            ))
    if not out:
        raise ValueError("No JWK rows found.")
    return out


def run_test1_parameterisation_integrity(rows: List[JWK], config: Config) -> Tuple[List[Dict[str, object]], Dict[str, object]]:
    results: List[Dict[str, object]] = []
    pass_count = 0
    amber_count = 0
    fail_count = 0

    for jwk in rows:
        raw, diag = jwk_to_raw_svi(jwk, config)
        base = {
            "case_id": jwk.case_id,
            "slice_id": jwk.slice_id,
            "tenor": jwk.tenor,
            "nu": jwk.nu,
            "psi": jwk.psi,
            "kappa": jwk.kappa,
            "p": jwk.p,
            "c": jwk.c,
            "conversion_status": diag.get("conversion_status", ""),
            "conversion_reason": diag.get("reason", ""),
            "beta": diag.get("beta", ""),
            "gamma": diag.get("gamma", ""),
            "raw_a": "",
            "raw_b": "",
            "raw_rho": diag.get("rho", ""),
            "raw_m": "",
            "raw_sigma": "",
            "raw_domain_status": "",
            "raw_domain_details": "",
            "roundtrip_max_abs_error": "",
            "roundtrip_status": "",
            "overall_status": "",
        }

        if raw is None:
            base["overall_status"] = "FAIL"
            fail_count += 1
            results.append(base)
            continue

        domain = classify_raw_svi(raw, config.boundary_tol)
        base.update({
            "raw_a": raw.a,
            "raw_b": raw.b,
            "raw_rho": raw.rho,
            "raw_m": raw.m,
            "raw_sigma": raw.sigma,
            "raw_domain_status": domain["domain_status"],
            "raw_domain_details": domain["details"],
        })

        if domain["domain_status"] == DOMAIN_OUTSIDE:
            base["overall_status"] = "FAIL"
            base["roundtrip_status"] = "NOT_ATTEMPTED_OUTSIDE_DOMAIN"
            fail_count += 1
            results.append(base)
            continue

        reverse, rev_diag = raw_to_jwk_consistent(raw, jwk.case_id, jwk.slice_id, jwk.tenor, config)
        if reverse is None:
            if diag.get("conversion_status") == "BOUNDARY_DEGENERATE":
                base["roundtrip_status"] = "BOUNDARY_DEGENERATE"
                base["overall_status"] = "AMBER"
                amber_count += 1
            else:
                base["roundtrip_status"] = rev_diag.get("reverse_status", "FAIL")
                base["overall_status"] = "FAIL"
                fail_count += 1
            results.append(base)
            continue

        diffs = [
            abs(reverse.nu - jwk.nu),
            abs(reverse.psi - jwk.psi),
            abs(reverse.kappa - jwk.kappa),
            abs(reverse.p - jwk.p),
            abs(reverse.c - jwk.c),
        ]
        max_diff = max(diffs)
        base["roundtrip_max_abs_error"] = max_diff
        base["roundtrip_status"] = "PASS" if max_diff <= config.roundtrip_tol else "FAIL"

        if base["roundtrip_status"] == "PASS" and domain["domain_status"] == DOMAIN_REGULAR:
            base["overall_status"] = "PASS"
            pass_count += 1
        elif base["roundtrip_status"] == "PASS" and domain["domain_status"] == DOMAIN_AT_BOUNDARY:
            base["overall_status"] = "AMBER"
            amber_count += 1
        else:
            base["overall_status"] = "FAIL"
            fail_count += 1

        results.append(base)

    summary = {
        "test_name": "parameterisation_integrity_and_admissible_domain_equivalence",
        "rows_checked": len(results),
        "pass_count": pass_count,
        "amber_count": amber_count,
        "fail_count": fail_count,
        "overall_status": "FAIL" if fail_count else ("AMBER" if amber_count else "PASS"),
    }
    return results, summary


def _interp_jwk(j0: JWK, j1: JWK, T: float, rule: str) -> JWK:
    if j1.tenor <= j0.tenor:
        raise ValueError("Tenors must be strictly increasing within each case_id")
    lam = (T - j0.tenor) / (j1.tenor - j0.tenor)

    def lin(x, y):
        return (1.0 - lam) * x + lam * y

    if rule == "linear_jwk":
        nu = lin(j0.nu, j1.nu)
    elif rule == "linear_sqrt_nu":
        nu = lin(math.sqrt(j0.nu), math.sqrt(j1.nu)) ** 2
    else:
        raise ValueError("interpolation_rule must be 'linear_jwk' or 'linear_sqrt_nu'")

    return JWK(
        case_id=j0.case_id,
        slice_id=f"interp_{j0.slice_id}_{j1.slice_id}_{T:.8f}",
        tenor=T,
        nu=nu,
        psi=lin(j0.psi, j1.psi),
        kappa=lin(j0.kappa, j1.kappa),
        p=lin(j0.p, j1.p),
        c=lin(j0.c, j1.c),
    )


def run_test2_calendar_interpolation_preservation(rows: List[JWK], config: Config) -> Tuple[List[Dict[str, object]], Dict[str, object]]:
    by_case: Dict[str, List[JWK]] = {}
    for r in rows:
        by_case.setdefault(r.case_id, []).append(r)

    k_grid = np.linspace(config.k_min, config.k_max, config.k_points)
    results: List[Dict[str, object]] = []
    fail_count = 0
    amber_count = 0
    pass_count = 0

    for case_id, case_rows in sorted(by_case.items()):
        case_rows = sorted(case_rows, key=lambda x: x.tenor)
        if len(case_rows) < 2:
            results.append({
                "case_id": case_id,
                "pillar_pair": "",
                "status": "INCOMPLETE",
                "comment": "At least two tenors required for calendar interpolation test",
                "worst_decrement": "",
                "worst_k": "",
                "worst_t_prev": "",
                "worst_t_curr": "",
            })
            amber_count += 1
            continue

        for left, right in zip(case_rows[:-1], case_rows[1:]):
            t_grid = np.linspace(left.tenor, right.tenor, config.t_interpolation_points)
            w_rows: List[np.ndarray] = []
            conversion_failed = False
            fail_reason = ""

            for T in t_grid:
                jwk_interp = _interp_jwk(left, right, float(T), config.interpolation_rule)
                raw, diag = jwk_to_raw_svi(jwk_interp, config)
                if raw is None:
                    conversion_failed = True
                    fail_reason = str(diag.get("reason", "conversion failed"))
                    break
                domain = classify_raw_svi(raw, config.boundary_tol)
                if domain["domain_status"] == DOMAIN_OUTSIDE:
                    conversion_failed = True
                    fail_reason = f"raw outside domain: {domain['details']}"
                    break
                w_rows.append(raw_svi_total_variance(k_grid, raw))

            if conversion_failed:
                results.append({
                    "case_id": case_id,
                    "pillar_pair": f"{left.slice_id}->{right.slice_id}",
                    "status": "FAIL",
                    "comment": fail_reason,
                    "worst_decrement": "",
                    "worst_k": "",
                    "worst_t_prev": "",
                    "worst_t_curr": "",
                })
                fail_count += 1
                continue

            w = np.vstack(w_rows)
            increments = np.diff(w, axis=0)
            min_idx = np.unravel_index(int(np.argmin(increments)), increments.shape)
            worst_decrement = float(increments[min_idx])
            worst_t_prev = float(t_grid[min_idx[0]])
            worst_t_curr = float(t_grid[min_idx[0] + 1])
            worst_k = float(k_grid[min_idx[1]])

            if worst_decrement < -config.calendar_tol:
                status = "FAIL"
                fail_count += 1
                comment = "calendar total-variance decrement detected between pillars"
            elif worst_decrement < 10.0 * config.calendar_tol:
                status = "AMBER"
                amber_count += 1
                comment = "near-zero calendar margin; monitor sensitivity"
            else:
                status = "PASS"
                pass_count += 1
                comment = "OK"

            results.append({
                "case_id": case_id,
                "pillar_pair": f"{left.slice_id}->{right.slice_id}",
                "status": status,
                "comment": comment,
                "worst_decrement": worst_decrement,
                "worst_k": worst_k,
                "worst_t_prev": worst_t_prev,
                "worst_t_curr": worst_t_curr,
            })

    summary = {
        "test_name": "surface_calendar_interpolation_preservation",
        "pillar_pairs_checked": len(results),
        "pass_count": pass_count,
        "amber_count": amber_count,
        "fail_count": fail_count,
        "overall_status": "FAIL" if fail_count else ("AMBER" if amber_count else "PASS"),
    }
    return results, summary


def write_csv(path: str, rows: List[Dict[str, object]]) -> None:
    if not rows:
        raise ValueError(f"No rows to write for {path}")
    with open(path, "w", newline="", encoding="utf-8") as f:
        writer = csv.DictWriter(f, fieldnames=list(rows[0].keys()))
        writer.writeheader()
        writer.writerows(rows)


def parse_args() -> argparse.Namespace:
    p = argparse.ArgumentParser(description="Run SVI-JWK theoretical conceptual soundness tests.")
    p.add_argument("--input", required=True, help="Input CSV of JWK parameter slices.")
    p.add_argument("--output-dir", required=True, help="Directory for result CSV/JSON files.")
    p.add_argument("--kappa-convention", choices=["raw", "stored_div_sqrtT"], default="raw")
    p.add_argument("--interpolation-rule", choices=["linear_jwk", "linear_sqrt_nu"], default="linear_jwk")
    p.add_argument("--k-min", type=float, default=-3.0)
    p.add_argument("--k-max", type=float, default=3.0)
    p.add_argument("--k-points", type=int, default=301)
    p.add_argument("--t-interpolation-points", type=int, default=25)
    p.add_argument("--calendar-tol", type=float, default=1e-10)
    p.add_argument("--roundtrip-tol", type=float, default=1e-8)
    return p.parse_args()


def main() -> int:
    args = parse_args()
    os.makedirs(args.output_dir, exist_ok=True)

    config = Config(
        kappa_convention=args.kappa_convention,
        interpolation_rule=args.interpolation_rule,
        k_min=args.k_min,
        k_max=args.k_max,
        k_points=args.k_points,
        t_interpolation_points=args.t_interpolation_points,
        calendar_tol=args.calendar_tol,
        roundtrip_tol=args.roundtrip_tol,
    )

    rows = load_jwk_csv(args.input)

    test1_rows, test1_summary = run_test1_parameterisation_integrity(rows, config)
    test2_rows, test2_summary = run_test2_calendar_interpolation_preservation(rows, config)

    test1_path = os.path.join(args.output_dir, "test1_parameterisation_integrity.csv")
    test2_path = os.path.join(args.output_dir, "test2_calendar_interpolation.csv")
    summary_path = os.path.join(args.output_dir, "summary.json")

    write_csv(test1_path, test1_rows)
    write_csv(test2_path, test2_rows)

    statuses = [test1_summary["overall_status"], test2_summary["overall_status"]]
    if "FAIL" in statuses:
        overall = "FAIL"
    elif "AMBER" in statuses:
        overall = "AMBER"
    else:
        overall = "PASS"

    summary = {
        "overall_status": overall,
        "input_file": os.path.abspath(args.input),
        "output_dir": os.path.abspath(args.output_dir),
        "config": config.__dict__,
        "tests": [test1_summary, test2_summary],
        "interpretation_note": (
            "FAIL indicates a theoretical domain or surface-construction issue under the tested convention. "
            "AMBER indicates a boundary, degeneracy, or near-zero margin requiring formal model limitation or monitoring."
        ),
    }

    with open(summary_path, "w", encoding="utf-8") as f:
        json.dump(summary, f, indent=2)

    print(json.dumps(summary, indent=2))
    return 0 if overall == "PASS" else 2
